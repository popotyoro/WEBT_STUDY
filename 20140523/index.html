Multiset< String > wordsSet = HashMultiset.create();
wordsSet.add("Suzuki");
wordsSet.add("Sato");
wordsSet.add("Takahashi");
wordsSet.add("Suzuki");

　Table


Table< String, String, String > nameTable = HashBasedTable.create();

nameTable.put("Man", "Suzuki", "Kosuke");
nameTable.put("Woman", "Suzuki", "Kouko");
nameTable.put("Man", "Sato", "Toshi");
System.out.println(nameTable.row("Man"));
System.out.println(nameTable.column("Suzuki"));
System.out.println(nameTable.row("Man").get("Sato"));

　実行結果


{Suzuki=Kosuke, Sato=Toshi}
{Man=Kosuke, Woman=Kouko}
Toshi
